可以用来作为js分页程序中显示页码部分的逻辑

```` javascript
/*
* 获得序列中的范围边界
* @param {Number} cl count length总长度
* @param {Number} p position 当前位置，默认为0
* @param {Number} l length 截取的长度，默认为4，最好为偶数
* @reutrn
*/

var getRangeSite = function(cl,p,l){
  var p = p || 0,l = l || 4, lp = p-l/2 ,rp = lp+l;
  if( lp<0 ){ rp = rp -lp; lp =0; }
  if( rp>cl ){ lp= lp-(rp-cl); lp= lp<0 ? 0 : lp ;rp=cl;}
  return [lp,rp]

}

````
